# basic set

>>> from botl.obj import Object
>>> o = Object()
>>> o.set("key", "value")
>>> o.key
'value'

# attribute set

>>> from botl.obj import Object
>>> o = Object()
>>> o.key = "value"
>>> o.key
'value'

# save/load

>>> import botl.obj
>>> botl.obj.wd = ".test"
>>> o = botl.obj.Object()
>>> o["key"] = "value"
>>> p = o.save()
>>> p
'botl.obj.Object/...
>>> oo = botl.obj.Object()
>>> oo.load(p)
>>> oo.key
'value'
